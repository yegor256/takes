24  package org.takes.facets.auth.codecs;
25  
26  import java.io.IOException;
27  import org.takes.facets.auth.Identity;
28  
29  /**
30   * Decorator which check incoming and outgoing identities.
31   *
32   * <p>The class is immutable and thread-safe.
33   * @author Dmitry Zaytsev (dmitry.zaytsev@gmail.com)
34   * @version $Id$
35   * @since 0.11.2
36   */
37  public final class CcStrict implements Codec {
38      /**
39       * Original codec.
40       */
41      private final transient Codec origin;
42  
43      /**
44       * Ctor.
45       * @param codec Original codec
46       */
47      public CcStrict(final Codec codec) {
48          this.origin = codec;
49      }
50  
51      @Override
52      public byte[] encode(final Identity identity) throws IOException {
53          return this.origin.encode(CcStrict.applyRules(identity));
54      }
55  
56      @Override
57      public Identity decode(final byte[] bytes) throws IOException {
58          return CcStrict.applyRules(this.origin.decode(bytes));
59      }
60  
61      /**
62       * Apply validation rules to identity.
63       * @param identity Identity
64       * @return Identity
65       */
66      private static Identity applyRules(final Identity identity) {
67          if (identity != Identity.ANONYMOUS) {
68              if (identity.urn().isEmpty()) {
69                  throw new DecodingException("urn is empty");
70              }
71              // @checkstyle LineLength (1 line)
72              if (!identity.urn().matches("^(?i)^urn(?-i):[a-zA-Z0-9]([\\-a-zA-Z0-9]{1,31})(:([\\-a-zA-Z0-9/]|%[0-9a-fA-F]{2})*)+(\\?\\w+(=([\\-a-zA-Z0-9/]|%[0-9a-fA-F]{2})*)?(&\\w+(=([\\-a-zA-Z0-9/]|%[0-9a-fA-F]{2})*)?)*)?\\*?$")) {
73                  throw new DecodingException("urn isn't valid");
74              }
75          }
76          return identity;
77      }
78  }
