46  /**
47   * Linkedin OAuth landing/callback page.
48   *
49   * <p>The class is immutable and thread-safe.
50   * @author Dmitry Zaytsev (dmitry.zaytsev@gmail.com)
51   * @version $Id: 47d2c55dfacaf999d12d9f670b5f27817e6b28a6 $
52   * @since 0.11.3
53   */
54  @EqualsAndHashCode(of = { "app", "key" })
55  public final class PsLinkedin implements Pass {
56  
57      /**
58       * App name.
59       */
60      private final transient String app;
61  
62      /**
63       * Key.
64       */
65      private final transient String key;
66  
67      /**
68       * Linkedin token href.
69       */
70      private final transient Href tkhref;
71  
72      /**
73       * Linkedin api href.
74       */
75      private final transient Href apihref;
76  
77      /**
78       * Ctor.
79       * @param lapp Linkedin app
80       * @param lkey Linkedin key
81       */
82      public PsLinkedin(final String lapp, final String lkey) {
83          this(
84              new Href("https://www.linkedin.com/uas/oauth2/accessToken"),
85              // @checkstyle LineLength (1 line)
86              new Href("https://api.linkedin.com/v1/people/~:(id,first-name,last-name,picture-url)"),
87              lapp,
88              lkey
89          );
90      }
91  
92      /**
93       * Ctor with custom hrefs for test purposes.
94       * @param thref Linkedin token href
95       * @param ahref Linkedin api href
96       * @param lapp Linkedin app name
97       * @param lkey Linkedin key
98       * @checkstyle ParameterNumberCheck (3 lines)
99       */
100     public PsLinkedin(final Href thref, final Href ahref,
101             final String lapp, final String lkey) {
102         this.tkhref = thref;
103         this.apihref = ahref;
104         this.app = lapp;
105         this.key = lkey;
106     }
107 
108     @Override
109     public Opt<Identity> enter(final Request request)
110         throws IOException {
111         final Href href = new RqHref.Base(request).href();
112         // @checkstyle MultipleStringLiteralsCheck (1 line)
113         final Iterator<String> code = href.param("code").iterator();
114         if (!code.hasNext()) {
115             throw new HttpException(
116                 HttpURLConnection.HTTP_BAD_REQUEST,
117                 "code is not provided by LinkedIn"
118             );
119         }
120         return new Opt.Single<Identity>(
121             this.fetch(this.token(href.toString(), code.next()))
122         );
123     }
124 
125     @Override
126     public Response exit(final Response response,
127         final Identity identity) {
128         return response;
129     }
130 
131     /**
132      * Get user name from Linkedin, with the token provided.
133      * @param token PsLinkedin access token
134      * @return The user found in PsLinkedin
135      * @throws IOException If fails
136      */
137     private Identity fetch(final String token) throws IOException {
138         // @checkstyle LineLength (1 line)
139         final String uri = this.apihref
140             .with("oauth2_access_token", token)
141             .with("format", "json")
142             .toString();
143         return PsLinkedin.parse(
144             new JdkRequest(uri)
145                 .header("accept", "application/json")
146                 .fetch().as(RestResponse.class)
147                 .assertStatus(HttpURLConnection.HTTP_OK)
148                 .as(JsonResponse.class).json().readObject()
149         );
150     }
151 
152     /**
153      * Retrieve PsLinkedin access token.
154      * @param home Home of this page
155      * @param code PsLinkedin "authorization code"
156      * @return The token
157      * @throws IOException If failed
158      */
159     private String token(final String home, final String code)
160         throws IOException {
161         final String uri = this.tkhref.toString();
162         return new JdkRequest(uri)
163             .method("POST")
164             .header("Accept", "application/xml")
165             .body()
166             .formParam("grant_type", "authorization_code")
167             .formParam("client_id", this.app)
168             .formParam("redirect_uri", home)
169             .formParam("client_secret", this.key)
170             .formParam("code", code)
171             .back()
172             .fetch().as(RestResponse.class)
173             .assertStatus(HttpURLConnection.HTTP_OK)
174             .as(JsonResponse.class)
175             .json().readObject().getString("access_token");
176     }
177 
178     /**
179      * Make identity from JSON object.
180      * @param json JSON received from Github
181      * @return Identity found
182      */
183     private static Identity parse(final JsonObject json) {
184         final String fname = "firstName";
185         final String lname = "lastName";
186         final String unknown = "?";
187         final ConcurrentMap<String, String> props =
188             new ConcurrentHashMap<String, String>(json.size());
189         props.put(fname, json.getString(fname, unknown));
190         props.put(lname, json.getString(lname, unknown));
191         return new Identity.Simple(
192             String.format("urn:linkedin:%s", json.getString("id")), props
193         );
194     }
