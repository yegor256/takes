24  package org.takes;
25  
26  import java.io.UnsupportedEncodingException;
27  import java.net.URI;
28  import java.net.URLDecoder;
29  import java.net.URLEncoder;
30  import java.nio.charset.Charset;
31  import java.util.Collections;
32  import java.util.LinkedList;
33  import java.util.List;
34  import java.util.Map;
35  import java.util.concurrent.ConcurrentHashMap;
36  import java.util.concurrent.ConcurrentMap;
37  import org.takes.misc.VerboseIterable;
38  
39  /**
40   * HTTP URI/HREF.
41   *
42   * <p>The class is immutable and thread-safe.
43   *
44   * @author Yegor Bugayenko (yegor@teamed.io)
45   * @version $Id$
46   * @since 0.7
47   */
48  public final class Href {
49  
50      /**
51       * URI (without the query part).
52       */
53      private final transient URI uri;
54  
55      /**
56       * Params.
57       */
58      private final transient ConcurrentMap<String, List<String>> params;
59  
60      /**
61       * Ctor.
62       * @param txt Text of the link
63       */
64      @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
65      public Href(final String txt) {
66          this.params = new ConcurrentHashMap<String, List<String>>(0);
67          final URI link = URI.create(txt);
68          final String query = link.getQuery();
69          if (query == null) {
70              this.uri = link;
71          } else {
72              final String href = link.toString();
73              this.uri = URI.create(
74                  href.substring(0, href.length() - query.length() - 1)
75              );
76              final String[] pairs = query.split("&");
77              for (final String pair : pairs) {
78                  final String[] parts = pair.split("=", 2);
79                  final String key = Href.decode(parts[0]);
80                  final String value;
81                  if (parts.length > 1) {
82                      value = Href.decode(parts[1]);
83                  } else {
84                      value = "";
85                  }
86                  this.params.putIfAbsent(key, new LinkedList<String>());
87                  this.params.get(key).add(value);
88              }
89          }
90      }
91  
92      /**
93       * Ctor.
94       * @param link The link
95       * @param map Map of params
96       */
97      private Href(final URI link,
98          final ConcurrentMap<String, List<String>> map) {
99          this.uri = link;
100         this.params = map;
101     }
102 
103     @Override
104     public String toString() {
105         final StringBuilder text = new StringBuilder(this.uri.toString());
106         if (!this.params.isEmpty()) {
107             boolean first = true;
108             for (final Map.Entry<String, List<String>> ent
109                 : this.params.entrySet()) {
110                 for (final String value : ent.getValue()) {
111                     if (first) {
112                         text.append('?');
113                         first = false;
114                     } else {
115                         text.append('&');
116                     }
117                     text.append(Href.encode(ent.getKey()))
118                         .append('=')
119                         .append(Href.encode(value));
120                 }
121             }
122         }
123         return text.toString();
124     }
125 
126     /**
127      * Get path part of the HREF.
128      * @return Path
129      * @since 0.9
130      */
131     public String path() {
132         return this.uri.getPath();
133     }
134 
135     /**
136      * Get query param.
137      * @param key Param name
138      * @return Values (could be empty)
139      * @since 0.9
140      */
141     public Iterable<String> param(final String key) {
142         final List<String> values = this.params.get(key);
143         final Iterable<String> iter;
144         if (values == null) {
145             iter = new VerboseIterable<String>(
146                 Collections.<String>emptyList(),
147                 String.format(
148                     "there are no URI params by name \"%s\" among %d others",
149                     key, this.params.size()
150                 )
151             );
152         } else {
153             iter = new VerboseIterable<String>(
154                 values,
155                 String.format(
156                     "there are only %d URI params by name \"%s\"",
157                     values.size(), key
158                 )
159             );
160         }
161         return iter;
162     }
163 
164     /**
165      * Add this path to the URI.
166      * @param suffix The suffix
167      * @return New HREF
168      */
169     public Href path(final String suffix) {
170         return new Href(
171             URI.create(
172                 new StringBuilder(this.uri.toString())
173                     .append(Href.encode(suffix)).toString()
174             ),
175             this.params
176         );
177     }
178 
179     /**
180      * Add this extra param.
181      * @param key Key of the param
182      * @param value The value
183      * @return New HREF
184      */
185     public Href with(final String key, final String value) {
186         final ConcurrentMap<String, List<String>> map =
187             new ConcurrentHashMap<String, List<String>>(
188                 this.params.size() + 1
189             );
190         map.putAll(this.params);
191         map.putIfAbsent(key, new LinkedList<String>());
192         map.get(key).add(value);
193         return new Href(this.uri, map);
194     }
195 
196     /**
197      * Without this query param.
198      * @param key Key of the param
199      * @return New HREF
200      */
201     public Href without(final String key) {
202         final ConcurrentMap<String, List<String>> map =
203             new ConcurrentHashMap<String, List<String>>(
204                 this.params.size()
205             );
206         map.putAll(this.params);
207         map.remove(key);
208         return new Href(this.uri, map);
209     }
210 
211     /**
212      * Encode into URL.
213      * @param txt Text
214      * @return Encoded
215      */
216     private static String encode(final String txt) {
217         try {
218             return URLEncoder.encode(
219                 txt, Charset.defaultCharset().name()
220             );
221         } catch (final UnsupportedEncodingException ex) {
222             throw new IllegalStateException(ex);
223         }
224     }
225 
226     /**
227      * Decode from URL.
228      * @param txt Text
229      * @return Decoded
230      */
231     private static String decode(final String txt) {
232         try {
233             return URLDecoder.decode(
234                 txt, Charset.defaultCharset().name()
235             );
236         } catch (final UnsupportedEncodingException ex) {
237             throw new IllegalStateException(ex);
238         }
239     }
