34  import java.util.AbstractMap;
35  import java.util.Map;
36  import java.util.concurrent.ConcurrentHashMap;
37  import java.util.concurrent.ConcurrentMap;
38  import lombok.EqualsAndHashCode;
39  import org.apache.velocity.VelocityContext;
40  import org.apache.velocity.app.VelocityEngine;
41  import org.apache.velocity.runtime.RuntimeConstants;
42  import org.apache.velocity.runtime.log.NullLogChute;
43  import org.takes.Response;

45  /**
46   * Response that converts Velocity template to text.
47   *
48   * <p>This response implementation is rendering a page from
49   * Apache Velocity template. Here is how you can use it:
50   *
51   * <pre>public final class TkHelp implements Take {
52   *   &#64;Override
53   *   public Response act(final Request req) {
54   *     return new RsHTML(
55   *       new RsVelocity(
56   *         this.getClass().getResource("help.html.vm"),
57   *         new RsVelocity.Pair("name", "Jeffrey")
58   *       )
59   *     );
60   *   }
61   * }</pre>
62   *
63   * <p>The class is immutable and thread-safe.
64   *
65   * @author Yegor Bugayenko (yegor@teamed.io)
66   * @version $Id$
67   * @since 0.1
68   * @checkstyle ClassDataAbstractionCouplingCheck (500 lines)
69   */
70  @EqualsAndHashCode(callSuper = true)
71  public final class RsVelocity extends RsWrap {
72  
73      /**
74       * Ctor.
75       * @param template Template
76       * @param params List of params
77       * @since 0.11
78       */
79      public RsVelocity(final CharSequence template,
80          final RsVelocity.Pair... params) {
81          this(new ByteArrayInputStream(template.toString().getBytes()), params);
82      }
83  
84      /**
85       * Ctor.
86       * @param template Template
87       * @param params List of params
88       * @throws IOException If fails
89       * @since 0.11
90       */
91      public RsVelocity(final URL template,
92          final RsVelocity.Pair... params) throws IOException {
93          this(template.openStream(), params);
94      }
95  
96      /**
97       * Ctor.
98       * @param template Template
99       * @param params Entries
100      */
101     public RsVelocity(final InputStream template,
102         final RsVelocity.Pair... params) {
103         this(template, RsVelocity.asMap(params));
104     }
105 
106     /**
107      * Ctor.
108      * @param tpl Template
109      * @param params Map of params
110      */
111     public RsVelocity(final InputStream tpl, final Map<CharSequence,
112         Object> params) {
113         super(
114             new Response() {
115                 @Override
116                 public Iterable<String> head() {
117                     return new RsEmpty().head();
118                 }
119                 @Override
120                 public InputStream body() throws IOException {
121                     return RsVelocity.render(tpl, params);
122                 }
123             }
124         );
125     }
126 
127     /**
128      * Render it.
129      * @param page Page template
130      * @param params Params for velocity
131      * @return Page body
132      * @throws IOException If fails
133      */
134     private static InputStream render(final InputStream page,
135         final Map<CharSequence, Object> params) throws IOException {
136         final ByteArrayOutputStream baos = new ByteArrayOutputStream();
137         final Writer writer = new OutputStreamWriter(baos);
138         final VelocityEngine engine = new VelocityEngine();
139         engine.setProperty(
140             RuntimeConstants.RUNTIME_LOG_LOGSYSTEM,
141             new NullLogChute()
142         );
143         engine.evaluate(
144             new VelocityContext(params),
145             writer,
146             "",
147             new InputStreamReader(page)
148         );
149         writer.close();
150         return new ByteArrayInputStream(baos.toByteArray());
151     }
152 
153     /**
154      * Convert entries to map.
155      * @param entries Entries
156      * @return Map
157      */
158     private static Map<CharSequence, Object> asMap(
159         final Map.Entry<CharSequence, Object>... entries) {
160         final ConcurrentMap<CharSequence, Object> map =
161             new ConcurrentHashMap<CharSequence, Object>(entries.length);
162         for (final Map.Entry<CharSequence, Object> ent : entries) {
163             map.put(ent.getKey(), ent.getValue());
164         }
165         return map;
166     }
167 
168     /**
169      * Pair of values.
170      */
171     public static final class Pair
172         extends AbstractMap.SimpleEntry<CharSequence, Object> {
173         /**
174          * Serialization marker.
175          */
176         private static final long serialVersionUID = 7362489770169963015L;
177         /**
178          * Ctor.
179          * @param key Key
180          * @param obj Pass
181          */
182         public Pair(final CharSequence key, final Object obj) {
183             super(key, obj);
184         }
185     }
186 }
