35  /**
36   * Test case for {@link RsPrettyXML}.
37   * @author Igor Khvostenkov (ikhvostenkov@gmail.com)
38   * @version $Id: 129626b134fd69ea2317a73724d52763cb877eb8 $
39   * @since 1.0
40   */
41  public final class RsPrettyXMLTest {
42  
43      /**
44       * RsPrettyXML can format response with XML body.
45       * @throws IOException If some problem inside
46       */
47      @Test
48      public void formatsXmlBody() throws IOException {
49          MatcherAssert.assertThat(
50              new RsPrint(
51                  new RsPrettyXML(
52                      new RsWithBody("<test><a>foo</a></test>")
53                  )
54              ).printBody(),
55              Matchers.is("<test>\n   <a>foo</a>\n</test>\n")
56          );
57      }
58  
59      /**
60       * RsPrettyXML can format HTML5 markup with proper DOCTYPE.
61       * @throws IOException If some problem inside
62       */
63      @Test
64      // @checkstyle MethodNameCheck (1 line)
65      public void formatsHtml5DoctypeBody() throws IOException {
66          MatcherAssert.assertThat(
67              new RsPrint(
68                  new RsPrettyXML(
69                      new RsWithBody(
70                          "<!DOCTYPE html><html><head></head><body></body></html>"
71                      )
72                  )
73              ).printBody(),
74              Matchers.containsString("<!DOCTYPE HTML>")
75          );
76      }
77  
78      /**
79       * RsPrettyXML can format HTML4 markup with DOCTYPE with public
80       * and system id.
81       * @throws IOException If some problem inside
82       */
83      @Test
84      // @checkstyle MethodNameCheck (1 line)
85      public void formatsHtml4DoctypeBody() throws IOException {
86          final String pid = "PUBLIC \"-//W3C//DTD HTML 4.01 Transitional//EN\" ";
87          final String xhtml = "<html xmlns=\"http://www.w3.org/1999/xhtml\" "
88              .concat("lang=\"en\">");
89          MatcherAssert.assertThat(
90              new RsPrint(
91                  new RsPrettyXML(
92                      new RsWithBody(
93                          Joiner.on("").appendTo(
94                              new StringBuilder("<!DOCTYPE HTML "),
95                              pid,
96                              "\"http://www.w3.org/TR/html4/loose.dtd\">",
97                              xhtml,
98                              "<head><a>foo</a></head>",
99                              "<body>this is body</body></html>"
100                         )
101                     )
102                 )
103             ).printBody(),
104             Matchers.is(
105                 Joiner.on("").appendTo(
106                     new StringBuilder("<!DOCTYPE html\n  "),
107                     pid,
108                     "\"http://www.w3.org/TR/html4/loose.dtd\">\n",
109                     xhtml,
110                     "\n   <head>\n      ",
111                     "<meta http-equiv=\"Content-Type\" content=\"text/html; ",
112                     "charset=UTF-8\" /><a>foo</a></head>\n   ",
113                     "<body>this is body</body>\n</html>"
114                 ).toString()
115             )
116         );
117     }
118 
119     /**
120      * RsPrettyXML can format response with non XML body.
121      * @throws IOException If some problem inside
122      */
123     @Test(expected = IOException.class)
124     public void formatsNonXmlBody() throws IOException {
125         new RsPrint(new RsPrettyXML(new RsWithBody("foo"))).printBody();
126     }
127 
128     /**
129      * RsPrettyXML can report correct content length.
130      * @throws IOException If some problem inside
131      */
132     @Test
133     public void reportsCorrectContentLength() throws IOException {
134         final ByteArrayOutputStream baos = new ByteArrayOutputStream();
135         new RsPrint(
136             new RsWithBody(
137                 "<test>\n   <a>test</a>\n</test>\n"
138             )
139         ).printBody(baos);
140         MatcherAssert.assertThat(
141             new RsPrint(
142                 new RsPrettyXML(
143                     new RsWithBody("<test><a>test</a></test>")
144                 )
145             ).printHead(),
146             Matchers.containsString(
147                 String.format(
148                     "Content-Length: %d",
149                     baos.toByteArray().length
150                 )
151             )
152         );
153     }
154 
155     /**
156      * RsPrettyXML can conform to equals and hash code contract.
157      * @throws Exception If some problem inside
158      */
159     @Test
160     public void conformsToEqualsAndHashCode() throws Exception {
161         EqualsVerifier.forClass(RsPrettyXML.class)
162             .suppress(Warning.TRANSIENT_FIELDS)
163             .withRedefinedSuperclass()
164             .verify();
165     }
