40  /**
41   * Test of {@link PsBasic}.
42   * @author Endrigo Antonini (teamed@endrigo.com.br)
43   * @version $Id: 50bc3f45344c2ae2574c9f60447a431466db6a14 $
44   * @since 0.20
45   */
46  public final class PsBasicTest {
47  
48      /**
49       * Basic Auth.
50       */
51      private static final String AUTH_BASIC = "Authorization: Basic %s";
52  
53      /**
54       * PsBasic can handle connection with valid credential.
55       * @throws Exception if any error occurs
56       */
57      @Test
58      public void handleConnectionWithValidCredential() throws Exception {
59          final String user = "john";
60          final Opt<Identity> identity = new PsBasic(
61              "RealmA",
62              new PsBasic.Fake(true)
63          ).enter(
64              new RqWithHeaders(
65                  new RqFake(
66                      RqMethod.GET,
67                      String.format(
68                          "?valid_code=%s",
69                          // @checkstyle MagicNumberCheck (1 line)
70                          RandomStringUtils.randomAlphanumeric(10)
71                      )
72                  ),
73                  this.generateAuthenticateHead(user, "pass")
74              )
75          );
76          MatcherAssert.assertThat(identity.has(), Matchers.is(true));
77          MatcherAssert.assertThat(
78              identity.get().urn(),
79              CoreMatchers.equalTo(this.generateIdentityUrn(user))
80          );
81      }
82  
83      /**
84       * PsBasic can handle connection with invalid credential.
85       * @throws Exception If some problem inside
86       */
87      @Test
88      public void handleConnectionWithInvalidCredential() throws Exception {
89          RsForward forward = new RsForward();
90          try {
91              new PsBasic(
92                  "RealmB",
93                  new PsBasic.Empty()
94              ).enter(
95                  new RqWithHeaders(
96                      new RqFake(
97                          RqMethod.GET,
98                          String.format(
99                              "?invalid_code=%s",
100                             // @checkstyle MagicNumberCheck (1 line)
101                             RandomStringUtils.randomAlphanumeric(10)
102                         )
103                     ),
104                     this.generateAuthenticateHead("username", "wrong")
105                 )
106             );
107         } catch (final RsForward ex) {
108             forward = ex;
109         }
110         MatcherAssert.assertThat(
111             new RsPrint(forward).printHead(),
112             Matchers.allOf(
113                 Matchers.containsString("HTTP/1.1 401 Unauthorized"),
114                 Matchers.containsString(
115                     "WWW-Authenticate: Basic ream=\"RealmB\""
116                 )
117             )
118         );
119     }
120 
121     /**
122      * PsBasic can handle multiple headers with valid credential.
123      * @throws Exception If some problem inside
124      */
125     @Test
126     public void handleMultipleHeadersWithValidCredential() throws Exception {
127         final String user = "bill";
128         final Opt<Identity> identity = new PsBasic(
129             "RealmC",
130             new PsBasic.Fake(true)
131         ).enter(
132             new RqWithHeaders(
133                 new RqFake(
134                     RqMethod.GET,
135                     String.format(
136                         "?multiple_code=%s",
137                         // @checkstyle MagicNumberCheck (1 line)
138                         RandomStringUtils.randomAlphanumeric(10)
139                     )
140                 ),
141                 this.generateAuthenticateHead(user, "changeit"),
142                 "Referer: http://teamed.io/",
143                 "Connection:keep-alive",
144                 "Content-Encoding:gzip",
145                 "X-Check-Cacheable:YES",
146                 "X-Powered-By:Java/1.7"
147             )
148         );
149         MatcherAssert.assertThat(identity.has(), Matchers.is(true));
150         MatcherAssert.assertThat(
151             identity.get().urn(),
152             CoreMatchers.equalTo(this.generateIdentityUrn(user))
153         );
154     }
155 
156     /**
157      * PsBasic can handle multiple headers with invalid content.
158      * @throws Exception If some problem inside
159      */
160     @Test(expected = HttpException.class)
161     public void handleMultipleHeadersWithInvalidContent() throws Exception {
162         MatcherAssert.assertThat(
163             new PsBasic(
164                 "RealmD",
165                 new PsBasic.Fake(true)
166             ).enter(
167                 new RqWithHeaders(
168                     new RqFake(
169                         "XPTO",
170                         "/wrong-url"
171                     ),
172                     String.format(
173                         "XYZ%s",
174                         this.generateAuthenticateHead("user", "password")
175                     ),
176                     "XYZReferer: http://teamed.io/",
177                     "XYZConnection:keep-alive",
178                     "XYZContent-Encoding:gzip",
179                     "XYZX-Check-Cacheable:YES",
180                     "XYZX-Powered-By:Java/1.7"
181                 )
182             ).has()
183             , Matchers.is(false)
184         );
185     }
186 
187     /**
188      * Generate the identity urn.
189      * @param user User
190      * @return URN
191      */
192     private String generateIdentityUrn(final String user) {
193         return String.format("urn:basic:%s", user);
194     }
195     /**
196      * Generate the string used on the request that store information about
197      * authentication.
198      * @param user Username
199      * @param pass Password
200      * @return Header string.
201      */
202     private String generateAuthenticateHead(
203         final String user,
204         final String pass
205     ) {
206         final String auth = String.format("%s:%s", user, pass);
207         final String encoded = DatatypeConverter.printBase64Binary(
208             auth.getBytes()
209         );
210         return String.format(AUTH_BASIC, encoded);
211     }
