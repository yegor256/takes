24  package org.takes.rq;
25  
26  import java.io.IOException;
27  import java.io.InputStream;
28  import java.util.Arrays;
29  import org.hamcrest.MatcherAssert;
30  import org.hamcrest.Matchers;
31  import org.junit.Test;
32  
33  /**
34   * Test case for {@link RqLengthAware}.
35   *
36   * @author Yegor Bugayenko (yegor@teamed.io)
37   * @version $Id: 6fb4ed5f6435e776eaa3f7495d1a1311b7b7ba60 $
38   * @since 0.1
39   */
40  public final class RqLengthAwareTest {
41  
42      /**
43       * RqLengthAware can add length to body.
44       * @throws IOException If some problem inside
45       */
46      @Test
47      public void addsLengthToBody() throws IOException {
48          MatcherAssert.assertThat(
49              new RqLengthAware(
50                  new RqFake(
51                      Arrays.asList(
52                          "GET /h?a=3",
53                          "Host: www.example.com",
54                          "Content-type: text/plain",
55                          "Content-Length: 2"
56                      ),
57                      "hi"
58                  )
59              ).body().available(),
60              Matchers.equalTo(2)
61          );
62      }
63  
64      /**
65       * RqLengthAware can add BIG length to body.
66       * @throws IOException If some problem inside
67       */
68      @Test
69      public void addsBigLengthToBody() throws IOException {
70          MatcherAssert.assertThat(
71              new RqLengthAware(
72                  new RqFake(
73                      Arrays.asList(
74                          "GET /hi-there",
75                          "Host: a.example.com",
76                          "Content-type: text/xml",
77                          "Content-Length: 9223372036854775000"
78                      ),
79                      "HI"
80                  )
81              ).body().available(),
82              Matchers.equalTo(Integer.MAX_VALUE)
83          );
84      }
85  
86      /**
87       * RqLengthAware can read byte.
88       * @throws IOException If some problem inside
89       */
90      @Test
91      public void readsByte() throws IOException {
92          final String data = "test";
93          final InputStream stream = new RqLengthAware(
94              new RqFake(
95                  Arrays.asList(
96                      "GET /test1",
97                      "Host: b.example.com"
98                  ),
99                  data
100             )
101         ).body();
102         MatcherAssert.assertThat(
103             stream.read(),
104             Matchers.equalTo((int) data.getBytes()[0])
105         );
106         MatcherAssert.assertThat(
107             stream.available(),
108             Matchers.equalTo(data.length() - 1)
109         );
110     }
111 
112     /**
113      * RqLengthAware can read byte array.
114      * @throws IOException If some problem inside
115      */
116     @Test
117     public void readsByteArray() throws IOException {
118         final String data = "array";
119         final InputStream stream = new RqLengthAware(
120             new RqFake(
121                 Arrays.asList(
122                     "GET /test2",
123                     "Host: c.example.com",
124                     "Content-type: text/csv"
125                 ),
126                 data
127             )
128         ).body();
129         final byte[] buf = new byte[data.length()];
130         MatcherAssert.assertThat(
131             stream.read(buf),
132             Matchers.equalTo(data.length())
133         );
134         MatcherAssert.assertThat(buf, Matchers.equalTo(data.getBytes()));
135         MatcherAssert.assertThat(stream.available(), Matchers.equalTo(0));
136     }
137 
138     /**
139      * RqLengthAware can read partial array.
140      * @throws IOException If some problem inside
141      */
142     @Test
143     public void readsPartialArray() throws IOException {
144         final String data = "hello world";
145         final int len = 3;
146         final InputStream stream = new RqLengthAware(
147             new RqFake(
148                 Arrays.asList(
149                     "GET /test3",
150                     "Host: d.example.com"
151                 ),
152                 data
153             )
154         ).body();
155         final byte[] buf = new byte[len];
156         MatcherAssert.assertThat(
157             stream.read(buf, 0, len),
158             Matchers.equalTo(len)
159         );
160         MatcherAssert.assertThat(
161             buf,
162             Matchers.equalTo(data.substring(0, len).getBytes())
163         );
164         MatcherAssert.assertThat(
165             stream.available(),
166             Matchers.equalTo(data.length() - len)
167         );
168     }
169 }
