178      * RqMultipart.Base can parse http body.
179      * @throws IOException If some problem inside
180      */
181     @Test
182     public void parsesHttpBody() throws IOException {
183         final RqMultipart multi = new RqMultipart.Fake(
184             new RqFake(),
185             new RqWithHeader(
186                 new RqFake("", "", "40 N Wolfe Rd, Sunnyvale, CA 94085"),
187                 DISPOSITION, "form-data; name=\"t4\""
188             ),
189             new RqWithHeader(
190                 new RqFake("", "", ""),
191                 DISPOSITION,
192                 "form-data; name=\"data\"; filename=\"a.bin\""
193             )
194         );
195         MatcherAssert.assertThat(
196             new RqHeaders.Base(
197                 multi.part("t4").iterator().next()
198             ).header(DISPOSITION),
199             Matchers.hasItem("form-data; name=\"t4\"")
200         );
201         MatcherAssert.assertThat(
202             new RqPrint(
203                 new RqHeaders.Base(
204                     multi.part("t4").iterator().next()
205                 )
206             ).printBody(),
207             Matchers.allOf(
208                 Matchers.startsWith("40 N"),
209                 Matchers.endsWith("CA 94085")
210             )
211         );
212     }

293     /**
294      * RqMultipart.Base can work in integration mode.
295      * @throws IOException if some problem inside
296      */
297     @Test
298     public void consumesHttpRequest() throws IOException {
299         final Take take = new Take() {
300             @Override
301             public Response act(final Request req) throws IOException {
302                 return new RsText(
303                     new RqPrint(
304                         new RqMultipart.Smart(
305                             new RqMultipart.Base(req)
306                         ).single("f-1")
307                     ).printBody()
308                 );
309             }
310         };
311         new FtRemote(take).exec(
312             // @checkstyle AnonInnerLengthCheck (50 lines)
313             new FtRemote.Script() {
314                 @Override
315                 public void exec(final URI home) throws IOException {
316                     new JdkRequest(home)
317                         .method("POST")
318                         .header(
319                             "Content-Type",
320                             "multipart/form-data; boundary=AaB0zz"
321                         )
322                         .body()
323                         .set(
324                             Joiner.on(RqMultipartTest.CRLF).join(
325                                 "--AaB0zz",
326                                 "Content-Disposition: form-data; name=\"f-1\"",
327                                 "",
328                                 "my picture",
329                                 "--AaB0zz--"
330                             )
331                         )
332                         .back()
333                         .fetch()
334                         .as(RestResponse.class)
335                         .assertStatus(HttpURLConnection.HTTP_OK)
336                         .assertBody(Matchers.containsString("pic"));
337                 }
338             }
339         );
340     }

342     /**
343      * RqMultipart.Base can handle a big request in an acceptable time.
344      * @throws IOException If some problem inside
345      */
346     @Test
347     @Category(PerformanceTests.class)
348     public void handlesRequestInTime() throws IOException {
349         final int length = 100000000;
350         final File temp = File.createTempFile("handlesRequestInTime", ".tmp");
351         final BufferedWriter bwr = new BufferedWriter(new FileWriter(temp));
352         bwr.write(
353             Joiner.on(RqMultipartTest.CRLF).join(
354                 "--zzz",
355                 "Content-Disposition: form-data; name=\"test\"",
356                 "",
357                 ""
358             )
359         );
360         for (int ind = 0; ind < length; ++ind) {
361             bwr.write("X");
362         }
363         bwr.write(RqMultipartTest.CRLF);
364         bwr.write("--zzz--");
365         bwr.write(RqMultipartTest.CRLF);
366         bwr.close();
367         final long start = System.currentTimeMillis();
368         final Request req = new RqFake(
369             Arrays.asList(
370                 "POST /post?u=3 HTTP/1.1",
371                 "Host: example.com",
372                 "Content-Type: multipart/form-data; boundary=zzz"
373             ),
374             new FileInputStream(temp)
375         );
376         MatcherAssert.assertThat(
377             new RqMultipart.Smart(
378                 new RqMultipart.Base(req)
379             ).single("test").body().available(),
380             Matchers.equalTo(length)
381         );
382         MatcherAssert.assertThat(
383             System.currentTimeMillis() - start,
384             //@checkstyle MagicNumberCheck (1 line)
385             Matchers.lessThan(3000L)
386         );
387         temp.delete();
388     }
389     /**
390      * RqMultipart.Base doesn't distort the content.
391      * @throws IOException If some problem inside
392      */
393     @Test
394     public void notDistortContent() throws IOException {
395         final int length = 1000000;
396         final File temp = File.createTempFile("notDistortContent", ".tmp");
397         final BufferedWriter bwr = new BufferedWriter(new FileWriter(temp));
398         bwr.write(
399             Joiner.on(RqMultipartTest.CRLF).join(
400                 "--zzz1",
401                 "Content-Disposition: form-data; name=\"test1\"",
402                 "",
403                 ""
404             )
405         );
406         final int byt = 0x7f;
407         for (int idx = 0; idx < length; ++idx) {
408             bwr.write(idx % byt);
409         }
410         bwr.write(RqMultipartTest.CRLF);
411         bwr.write("--zzz1--");
412         bwr.write(RqMultipartTest.CRLF);
413         bwr.close();
414         final Request req = new RqFake(
415             Arrays.asList(
416                 "POST /post?u=3 HTTP/1.1",
417                 "Host: exampl.com",
418                 "Content-Type: multipart/form-data; boundary=zzz1"
419             ),
420             new FileInputStream(temp)
421         );
422         final InputStream stream = new RqMultipart.Smart(
423             new RqMultipart.Base(req)
424         ).single("test1").body();
425         MatcherAssert.assertThat(
426             stream.available(),
427             Matchers.equalTo(length)
428         );
429         for (int idx = 0; idx < length; ++idx) {
430             MatcherAssert.assertThat(
431                 String.format("byte %d not matched", idx),
432                 stream.read(),
433                 Matchers.equalTo(idx % byt)
434             );
435         }
436         temp.delete();
437     }
